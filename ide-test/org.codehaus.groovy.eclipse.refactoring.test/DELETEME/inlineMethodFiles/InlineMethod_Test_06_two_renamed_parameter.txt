 * Inline with two renamed parameters
 */
class InlineSamples06 {
    static void main(args) {
        def a = 3, b = 7
        a = extract(a,b)
        println("a : " +a + " b: " + b)
    }
    static def extract(def b, def a) {
        println("a: " + a)
        return ++b
    }
}
