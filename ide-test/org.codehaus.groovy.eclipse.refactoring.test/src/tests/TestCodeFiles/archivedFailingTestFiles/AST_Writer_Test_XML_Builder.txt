###src
import groovy.xml.MarkupBuilder
def writer = new FileWriter('MyQuickWebSite .html')
def html = new MarkupBuilder(writer)
html.html {
    head {
        title 'RealQuick using MarkupBuilder'
    }
    body {
        h1 'Quickly printing a List of Elements'
        for ( line in ['Reading a book', 'Play Soccer', 'Fix the tests ']){
            h2( line )
        }
    }
}
